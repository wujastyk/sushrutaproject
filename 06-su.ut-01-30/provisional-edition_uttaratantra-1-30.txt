            <caesura/>

          <l xml:id="SS.6.1.17"><label>1938 ed. 6.1.17</label>
             dve vartmapaṭale vidyāc catvāry anyāni cākṣṇi tu |

          <l xml:id="SS.6.1.18"><label>1938 ed. 6.1.18</label>
             tejo jalāśritaṃ bāhyaṃ teṣv anyapiśitāśritaṃ |
            <caesura/>
             medas tṛtīyam paṭalam āśritaṃ tv asthi cāparaṃ |</l>

          <l xml:id="SS.6.1.19"><label>1938 ed. 6.1.19</label>
             pañcamāśaṃ samandṛṣṭes teṣāṃ bāhulyam iṣyate |
            <caesura/>
             sirāṇāṃ kaṇḍarāṇāñ ca medasaḥ kṛṣṇakasya ca </l>

          <l xml:id="SS.6.1.20"><label>1938 ed. 6.1.20</label>
             guṇā kṛṣṇātparaṃ śleṣmā bandhanekṣṇoḥ sirāyutaḥ |
            <caesura/>
             sirānusāribhir doṣair viguṇairūrdhvamāśritaiḥ |</l>

          <l xml:id="SS.6.1.21"><label>1938 ed. 6.1.21</label>
             jāyante netrabhāgeṣu vyādhayaś cāstu dāruṇāḥ |
            <caesura/>
             tatrāvilaṃ sasaṃrambhamaśrukaṇḍūpadehavat |</l>

          <l xml:id="SS.6.1.22"><label>1938 ed. 6.1.22</label>
             gurūṣācoparāgādyair juṣṭañcāvyaktalakṣaṇaiḥ |
            <caesura/>
             saśūlaṃ vartmakośeṣu śukapūrṇṇābham eva ca |</l>

          <l xml:id="SS.6.1.23"><label>1938 ed. 6.1.23</label>
             vihanyamānaṃ rūpeṣu kriyāsvakṣi yathāparaṃ |
            <caesura/>
             dṛṣṭvaivamdhīmāñjānīyāddoṣaiṇādhiṣṭhitantu tat |</l>

          <l xml:id="SS.6.1.24"><label>1938 ed. 6.1.24</label>
             tatra saṃcayamāsādya vidadhyāttatra bheṣajaṃ |</l>

          <l xml:id="SS.6.1.25"><label>1938 ed. 6.1.25</label>
             saṃkṣepataḥ kriyāyogo nidānaparivarjanaṃ |
            <caesura/>
             vātādīnāṃ pratīkāraḥprokto vistarataḥ purā |</l>

          <l xml:id="SS.6.1.26"><label>1938 ed. 6.1.26</label>
             uṣṇābhitaptasya jalapraveśāddūrekṣaṇāt svapnaviparyayācca |</l>

          <l xml:id="SS.6.1.27"><label>1938 ed. 6.1.27</label>
             svesadrajodhūmaniṣevaṇādvā chardobhighātādvamanātiyogāt |
            <caesura/>
             tathā dravānnātiniṣevaṇācca viṇmūtravātakramanigrahācca |
            <caesura/>

            <caesura/>

            <caesura/>
             bāṣpagrahātsūkṣmanirīkṣaṇācca netre vikārāñjanayanti doṣāḥ |</l>

          <l xml:id="SS.6.1.28"><label>1938 ed. 6.1.28</label>
             taistribhirdaśadaśa ca kaphenāpyadhikāstrayaḥ |

          <l xml:id="SS.6.1.29"><label>1938 ed. 6.1.29</label>

          <l xml:id="SS.6.1.37"><label>1938 ed. 6.1.37</label>
             manthasyandau kliṣṭavartma harṣotpattau tathaiva ca |
