          <l xml:id="SS.6.45.13"> atipravṛddhadoṣasya pūrvaṃ lohitapittinaḥ |
            <caesura/>

          <l xml:id="SS.6.45.14"> laṅghitasya tataḥ peyāṃ vidadhyāt svalpataṇḍulām |

            <caesura/>
             tarpaṇaṃ pācanaṃ lehān sarpīṣi vividhāni ca || </l>

          <l xml:id="SS.6.45.15"> drākṣāmadhukakāśmaryasitāyuktaṃ virecanam |

            <caesura/>

            <caesura/>

            <caesura/>

          <l xml:id="SS.6.45.20"> lihyāc ca dūrvāvaṭajāṃś ca pallavān madhudvitīyān

            <caesura/>
