        <div n="27" type="adhyāya">
          <head>saptaviṃśatitamo 'dhyāyaḥ | </head>
          <l xml:id="SS.6.27.1"> athāto navagrahākṛtivijñānīyam adhyāyaṃ vyākhyāsyāmaḥ ||</l>

          <l xml:id="SS.6.27.2"> yathovāca bhagavān dhanvantariḥ ||</l>

          <l xml:id="SS.6.27.3"> bālagrahāṇāṃ vijñānaṃ sādhanaṃ cāpy anantaram |
            <caesura/>
             utpattiṃ kāraṇaṃ caiva suśrutaikamanāḥ śṛṇu || </l>


          <l xml:id="SS.6.27.4"> skandagrahas tu prathamaḥ skandāpasmāra eva ca |
            <caesura/>
             śakunī revatī caiva pūtanā cāndhapūtanā || </l>


          <l xml:id="SS.6.27.5"> pūtanā śītanāmā ca tathaiva mukhamaṇḍikā |
            <caesura/>
             navamo naigameṣaś ca yaḥ pitṛgrahasaṃjñitaḥ || </l>


          <l xml:id="SS.6.27.6"> dhātrīmātroḥ prākpradiṣṭāpacārācchaucabhraṣṭānmaṅgalācārahīnān |
            <caesura/>
             trastān hṛṣṭāṃtarjitān tāḍitān vā pūjāhetorhiṃsyurete kumārān || </l>


          <l xml:id="SS.6.27.7"> aiśvaryasthāste na śakyā viśanto dehaṃ draṣṭuṃ mānuṣairviśvarūpāḥ |
            <caesura/>
             āptaṃ vākyaṃ tatsamīkṣyābhidhāsye liṅgānyeṣāṃ yāni dehe bhavanti || </l>


          <l xml:id="SS.6.27.8"> śūnākṣaḥ kṣatajasagandhikaḥ stanadviḍ vakrāsyo
            hatacalitaikapakṣmanetraḥ |
            <caesura/>
             udvigraḥ sululitacakṣuralparodī skandārto bhavati ca gāḍhamuṣṭivarcāḥ || </l>


          <l xml:id="SS.6.27.9"> niḥsaṃjño bhavati punar bhavet sasaṃjñaḥ saṃrabdhaḥ karacaraṇaiś ca
            nṛtyatīva |
            <caesura/>
             viṇmūtre sṛjati vinadya jṛmbhamāṇaḥ phenaṃ ca prasṛjati tat sakhābhipannaḥ || </l>


          <l xml:id="SS.6.27.10"> srastāṅgo bhayacakito vihaṅgagandhiḥ saṃsrāvivraṇaparipīḍitaḥ
            samantāt |
            <caesura/>
             sphoṭaiś ca pracitatanuḥ sadāhapākair vijñeyo bhavati śiśuḥ kṣataḥ śakunyā || </l>


          <l xml:id="SS.6.27.11"> raktāsyo haritamalo 'tipāṇḍudehaḥ śyāvo vā jvaramukhapākavedanātaḥ
            |
            <caesura/>
             revatyā viyathitatanuś ca karṇanāsaṃ mṛdgāti dhruvam abhipīḍitaḥ kumāraḥ || </l>


          <l xml:id="SS.6.27.12"> srastāṅgaḥ svapiti sukhaṃ divā na rātrau viḍ bhinnaṃ sṛjati ca
            kākatulyagandhiḥ |
            <caesura/>
             chardyārto hṛṣitatanūruhaḥ kumāras tṛṣṇālur bhavati ca pūtanāgṛhītaḥ || </l>


          <l xml:id="SS.6.27.13"> yo dveṣṭi stanamatisārakāsahikkācchardībhir jvarasahitābhir
            ardyamānaḥ |
            <caesura/>
             durvarṇaḥ satatam adhaḥśayo 'mlagandhistaṃ brūyur bhiṣaja ihāndhapūtanārtam || </l>


          <l xml:id="SS.6.27.14"> udvigro bhṛśam ativepate prarudyāt saṃlīnaḥ ssvapiti ca yasya
            cāntrakūjaḥ |
            <caesura/>
             visrāṅgo bhṛśam atisāryate ca yastaṃ jānīyād bhiṣag iha śītapūtanārtam || </l>


          <l xml:id="SS.6.27.15"> mlānāṅgaḥ surucir apāṇipādavaktro bahvāśī kaluṣasirāvṛtodaro yaḥ |
            <caesura/>
             sodvego bhavati ca mūtratulyagandhiḥ sa jñeyaḥ śiśur iha vaktramaṇḍikārtaḥ || </l>


          <l xml:id="SS.6.27.16"> yaḥ phenaṃ vamati vinamyate ca madhye sodvegaṃ vilapati
            cordhvamīkṣamāṇaḥ |
            <caesura/>
             jvaryeta pratatamatho vasāsagandhir niḥsaṃjño bhavati hi naigameṣajuṣṭaḥ || </l>


          <l xml:id="SS.6.27.17"> prastabdho yaḥ stanadveṣī muhyate cāviśanmuhuḥ |
            <caesura/>
             taṃ bālamacirāddhanti grahaḥ saṃpūrṇalakṣaṇaḥ || </l>


          <l xml:id="SS.6.27.18"> viparītamataḥ sādhyaṃ cikitsedacirārditam |
            <caesura/>
             gṛhe purāṇahaviṣābhyajya bālaṃ śucau śuciḥ || </l>


          <l xml:id="SS.6.27.19"> sarṣapān prakiretteṣāṃ tailairdīpaṃ ca kārayet |
            <caesura/>
             sadā sannihitaṃ cāpi juhuyāddhavyavāhanam || </l>


          <l xml:id="SS.6.27.20"> sarvagandhauṣadhībījairgandhamālyairalaṅkṛtam |
            <caesura/>
             agnye kṛttikābhyaś ca svāhā svāheti saṃtatam || </l>


          <l xml:id="SS.6.27.21"> namaḥ skandāya devāya grahādhipataye namaḥ |
            <caesura/>
             śirasā tvābhivande 'haṃ pratigṛhṇīṣva me balim |
            <caesura/>
             nīrujo nirvikāraś ca śiśurme jāyatāṃ drutam ||</l>


          <l xml:id="SS.6.27.trailer">iti suśrutasaṃhitāyām uttaratantrāntargate kumāratantre
            navagrahākṛtivijñānīyo nāma (prathamo 'dhyāyaḥ, āditaḥ) saptaviṃśo 'dhyāyaḥ ||27 ||</l>
        </div>


        <div n="28" type="adhyāya">
          <head>aṣṭāviṃśatitamo 'dhyāyaḥ | </head>
          <l xml:id="SS.6.28.1"> athātaḥ skandhagrahapratiṣedhaṃ vyākhyāsyāmaḥ ||</l>

          <l xml:id="SS.6.28.2"> yathovāca bhagavān dhanvantariḥ || </l>

          <l xml:id="SS.6.28.3"> skandagrahopasṛṣṭānāṃ kumārāṇāṃ praśasyate |
            <caesura/>
             vātaghnadrumapatrāṇāṃ niṣkvāthaḥ pariṣecane || </l>


          <l xml:id="SS.6.28.4"> teṣāṃ mūleṣu siddhaṃ ca tailamabhyañjane hitam |
            <caesura/>
             sarvagandhasurāmaṇḍakaiḍaryāvāpamiṣyate || </l>


          <l xml:id="SS.6.28.5"> devadāruṇi rāsnāyāṃ madhureṣu drumeṣu ca |
            <caesura/>
             siddhaṃ sarpiś ca sakṣīraṃ pānamasmai prayojayet || </l>


          <l xml:id="SS.6.28.6"> sarṣapāḥ sarpanirmoko vacā kākādanī ghṛtam |
            <caesura/>
             uṣṭrājāvigavāṃ caiva romāṇyuddhūpanaṃ śiśoḥ || </l>


          <l xml:id="SS.6.28.7"> somavallīmindravallīṃ śamīṃ bilvasya kaṇṭakān |
            <caesura/>
             mṛgādanyāś ca mūlāni grathitānyeva dhārayet || </l>


          <l xml:id="SS.6.28.8"> raktāni mālyāni tathā patākā raktāś ca gandhā vividhāś ca bhakṣyāḥ
            |
            <caesura/>
             ghaṇṭā ca devāya balirnivedyaḥ sukukkuṭaḥ skandagrahe hitāya || </l>


          <l xml:id="SS.6.28.9"> snānaṃ trirātraṃ niśi catvareṣu kuryāt puraṃ śāliyavairnavais tu |
            <caesura/>
             adbhiś ca gāyatryabhimantritābhiḥ prajvālanaṃ vyāhṛtibhiś ca vahneḥ || </l>


          <l xml:id="SS.6.28.10"> rakṣāmataḥ pravakṣyāmi bālānāṃ pāpanāśinīm |
            <caesura/>
             ahanyahani kartavyā yā bhiṣagbhir atandritaiḥ || </l>


          <l xml:id="SS.6.28.11"> tapasāṃ tejasāṃ caiva yaśasāṃ vapuṣāṃ tathā |
            <caesura/>
             nidhānaṃ yo 'vyayo devaḥ sa te skandaḥ prasīdatu || </l>


          <l xml:id="SS.6.28.12"> grahasenāpatirdevo devasenāpatirvibhuḥ |
            <caesura/>
             devasenātipuharaḥ pātu tvāṃ bhagavān guhaḥ || </l>


          <l xml:id="SS.6.28.13"> devadevasya mahataḥ pāvakasya ca yaḥ sutaḥ |
            <caesura/>
             gaṅgomākṛttikānāṃ ca sa te śarma prayacchatu || </l>


          <l xml:id="SS.6.28.14"> raktamālyāmbaraḥ śrīmān raktacandanabhūṣitaḥ |
            <caesura/>
             raktadivyavapurdevaḥ pātu tvāṃ krauñcasūdanaḥ ||</l>


          <l xml:id="SS.6.28.trailer">iti suśrutasaṃhitāyām uttaratantrāntargate kumāratantre
            skandapratiṣedho nāmā (dvitīyo 'dhyāyaḥ, ādito+)aṣṭāviṃśo 'dhyāyaḥ ||28 ||</l>
        </div>


        <div n="29" type="adhyāya">
          <head>ekonatriṃśattamo 'dhyāyaḥ | </head>
          <l xml:id="SS.6.29.1"> athātaḥ skandāpasmārapratiṣedhaṃ vyākhyāsyāmaḥ ||</l>

          <l xml:id="SS.6.29.2"> yathovāca bhagavān dhanvantariḥ ||</l>

          <l xml:id="SS.6.29.3"> bilvaḥ śirīṣo golomī surasādiś ca yo gaṇaḥ |
            <caesura/>
             pariṣeke prayoktavyaḥ skandāpasmāraśāntaye || </l>


          <l xml:id="SS.6.29.4"> sarvagandhavipakvaṃ tu tailamabhyañjane hitam |
            <caesura/>
             kṣīravṛkṣakaṣāye ca kākolyādau gaṇe tathā || </l>


          <l xml:id="SS.6.29.5ab"> vipaktavyaṃ ghṛtaṃ cāpi pānīyaṃ payasā saha |</l>
          <l xml:id="SS.6.29.5cd"> utsādanaṃ vacāhiṅguyuktaṃ skandagrahe hitam || </l>


          <l xml:id="SS.6.29.6"> gṛdhrolūkapurīṣāṇi keśā hastinakhā ghṛtam |
            <caesura/>
             vṛkṣabhasya ca romāṇi yojyāny uddhūpane 'pi ca || </l>


          <l xml:id="SS.6.29.7"> anantāṃ kukkuṭīṃ bimbīṃ markaṭīṃ cāpi dhārayet |
            <caesura/>
             pakvāpakvāni māṃsāni prasannā rudhiraṃ payaḥ || </l>


          <l xml:id="SS.6.29.8"> bhūtaudano nivedyaś ca skandāpasmāriṇe 'vaṭe |
            <caesura/>
             catuṣpathe ca kartavyaṃ snānamasya yatātmanā || </l>


          <l xml:id="SS.6.29.9"> skandāpasmārasaṃjño yaḥ skandasya dayitaḥ sakhā |
            <caesura/>
             viśākhasaṃjñaś ca śiśoḥ śivo 's tu vikṛtānanaḥ ||</l>


          <l xml:id="SS.6.29.trailer">iti suśrutasaṃhitāyām uttaratantrāntargate kumāratantre
            skandāpasmārapratiṣedho nāma (tṛtīyo 'dhyāyaḥ, āditaḥ) ekonatriṃśo 'dhyāyaḥ ||29 ||</l>
        </div>

          <l xml:id="SS.6.30.10"> antarīkṣacarā devī sarvālaṅkārabhūṣitā |
            <caesura/>

          <l xml:id="SS.6.30.11"> durdarśanā mahākāyā piṅgākṣī bhairavasvarā |
            <caesura/>
             lambodarī śaṅkukarṇī śakunī te prasīdatu ||</l>

          <l xml:id="SS.6.30.trailer">iti suśrutasaṃhitāyām uttaratantrāntargate kumāratantre
            śakunīpratiṣedho nāma (caturtho 'dhyāyaḥ, āditaḥ) triṃśo 'dhyāyaḥ || 30 ||</l>

        <div n="31" type="adhyāya">

          <head>ekatriṃśattamo 'dhyāyaḥ | </head>

          <l xml:id="SS.6.31.1"> athāto revatīpratiṣedhaṃ vyākhyāsyāmaḥ ||</l>

          <l xml:id="SS.6.31.2"> yathovāca bhagavān dhanvantariḥ ||</l>

          <l xml:id="SS.6.31.3"> aśvagandhā ca śṛṅgī ca sārivā sapunarnavā |
            <caesura/>
             sahe tathā vidārī ca kaṣāyāḥ secane hitāḥ || </l>

          <l xml:id="SS.6.31.4"> tailamabhyañjane kāryaṃ kuṣṭhe sarjarase 'pi ca |
            <caesura/>
             palaṅkaṣāyāṃ nalade tathā girikadambake || </l>

          <l xml:id="SS.6.31.5"> dhavāśvakarṇakakubhadhātakītindukīṣu ca |
            <caesura/>
             kākolyādigaṇe caiva pānīyaṃ sarpiriṣyate || </l>

          <l xml:id="SS.6.31.6ab"> kulatthāḥ śaṅkhacūrṇaṃ ca pradehaḥ sārvagandhikaḥ |</l>

          <l xml:id="SS.6.31.6cd"> gṛdhrolūkapurīṣāṇi yavā yavaphalo ghṛtam || </l>

          <l xml:id="SS.6.31.7"> sandhyayor ubhayoḥ kāryam etad uddhūpanaṃ śiśoḥ |
            <caesura/>
             varuṇāriṣṭakamayaṃ rucakaṃ saindukaṃ tathā || </l>

          <l xml:id="SS.6.31.8"> satataṃ dhārayec cāpi kṛtaṃ vā pautrajīvikam |
            <caesura/>
             śuklāḥ sumanaso lājāḥ payaḥ śālyodanaṃ tathā || </l>

          <l xml:id="SS.6.31.9"> balir nivedyo gotīrthe revatyai prayatātmanā |
            <caesura/>
             saṅgame ca bhiṣak snānaṃ kuryād dhātrīkumārayoḥ || </l>

          <l xml:id="SS.6.31.10"> nānāvastradharā devī citramālyānulepanā |
            <caesura/>
