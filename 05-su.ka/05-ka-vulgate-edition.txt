                    <l xml:id="SS.5.1.15"> saṃvibhaktāḥ sumanaso nīcakeśanakhāḥ sthirāḥ |

                    <l xml:id="SS.5.1.16"> tasya cājñāvidheyāḥ syur vividhāḥ parikarmiṇaḥ |

                    <l xml:id="SS.5.1.17ab"> tasmān mahānase vaidyaḥ pramādarahito bhavet |</l>

                    <l xml:id="SS.5.1.19"> vidyād viṣasya dātāram ebhir liṅgaiś ca buddhimān |

                    <l xml:id="SS.5.1.20"> apārthaṃ bahusaṅkīrṇaṃ bhāṣate cāpi mūḍhavat |

                    <l xml:id="SS.5.1.21"> vepathur jāyate tasya trastaś cānyo 'nyam īkṣate |
                        <caesura/>
                         kṣāmo vivarṇavaktraś ca nakhaiḥ kiñcic chinatty api || </l>

                    <l xml:id="SS.5.1.22"> ālabhetāsakṛddīnaḥ kareṇa ca śiroruhān |
                        <caesura/>

                    <l xml:id="SS.5.1.23"> vartate viparītaṃ tu viṣadātā vicetanaḥ |
                        <caesura/>
                         kecid bhayāt pārthivasya tvaritā vā tadājñayā || </l>

                    <l xml:id="SS.5.1.24"> asatām api santo 'pi ceṣṭāṃ kurvanti mānavāḥ |
                        <caesura/>
                         tasmāt parīkṣaṇaṃ kāryaṃ bhṛtyānām ādṛtair nṛpaiḥ || </l>

                    <l xml:id="SS.5.1.25"> anne pāne dantakāṣṭhe tathā 'bhyaṅge 'valekhane |

                    <l xml:id="SS.5.1.27"> viṣajuṣṭeṣu cānyeṣu nasyadhūmāñjanādiṣu |
                        <caesura/>

                    <l xml:id="SS.5.1.28"> nṛpabhaktād baliṃ nyastaṃ saviṣaṃ bhakṣayanti ye |
                        <caesura/>
                         tatraiva te vinaśyanti makṣikāvāyasādayaḥ || </l>

                    <l xml:id="SS.5.1.29"> hutabhuk tena cānnena bhṛśaṃ caṭacaṭāyate |
                        <caesura/>

                    <l xml:id="SS.5.1.30ab"> bhinnārcistīkṣṇadhūmaś ca nacirāccopaśāmyati |</l>

                    <l xml:id="SS.5.1.30cd"> cakorasyākṣivairāgyaṃ jāyate kṣipram eva tu || </l>

                    <l xml:id="SS.5.1.31"> dṛṣṭvā 'nnaṃ viṣasaṃsṛṣṭaṃ mriyante jīvajīvakāḥ |

                         kokilaḥ svaravaikṛtyaṃ krauñcas tu madam ṛcchati || </l>

                    <l xml:id="SS.5.1.32"> hṛṣyen mayūra udvignaḥ krośataḥ śukasārike |
                        <caesura/>

                    <l xml:id="SS.5.1.33ab"> pṛṣato visṛjaty aśruṃ viṣṭhāṃ muñcati markaṭaḥ |</l>

                    <l xml:id="SS.5.1.33cd"> sannikṛṣṭāṃs tataḥ kuryād rājñas tān mṛgapakṣiṇaḥ || </l>

                    <l xml:id="SS.5.1.34ab"> veśmano 'tha vibhūṣārthaṃ rakṣārthaṃ cātmanaḥ sadā | </l>
