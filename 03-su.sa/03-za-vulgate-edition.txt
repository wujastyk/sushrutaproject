        <head>śārīrasthānam</head>

        <div n="1" type="adhyāya">
          <head>prathamo 'dhyāyaḥ |</head>
          <l xml:id="SS.3.1.1"><label>śā.1.1</label> athātaḥ sarvabhūtacintāśārīraṃ vyākhyāsyāmaḥ || </l>

          <l xml:id="SS.3.1.2"><label>śā.1.2</label> yathovāca bhagavān dhanvantariḥ || </l>

          <l xml:id="SS.3.1.3"><label>śā.1.3</label> sarvabhūtānāṃ kāraṇam akāraṇaṃ
            sattvarajastamolakṣaṇam aṣṭarūpam akhilasya jagataḥ saṃbhavahetur avyaktaṃ nāma | tad
            ekaṃ <ref rend="superscript footnote marker">1</ref>bahūnāṃ<note type="footnote">
              'anekeṣāṃ 'iti pā+</note> kṣetrajñānām adhiṣṭhānaṃ samudra ivaudakānāṃ bhāvānām || </l>

          <l xml:id="SS.3.1.4"><label>śā.1.4</label> tasmād avyaktān mahān utpadyate talliṅga eva;
            talliṅgāc ca mahatas tallakṣaṇa evāhaṅkāra utpadyate, sa trividho vaikārikastaijaso
            bhūtādir iti; tatra vaikārikādahaṅkārāttaijasasahāyāt tallakṣaṇāny enaikādaśendriyāṇy
            utpadyante, tad yathā --- śrotratvakcakṣurjihvāghrāṇavāgghastopasthapāyupādamanāṃsīti,
            tatra pūrvāṇi pañca buddhīndriyāṇi, itarāṇi pañca karmendriyāṇi, ubhayātmakaṃ manaḥ;
            bhūtāder api taijasasahāyāt tallakṣaṇāny eva pañcatanmātrāṇy utpadyante, tad yathā ---
            śabdatanmātraṃ, sparśatanmātraṃ, rūpatanmātraṃ, rasatanmātraṃ, gandhatanmātram iti;
            teṣāṃ viśeṣāṃ viśeṣāḥ śabdasparśarūparasagandhāḥ; tebhyo bhūtāni vyomānilānalajalorvyaḥ;
            evam eṣā tattvacaturviṃśatir vyākhyātā || </l>

          <l xml:id="SS.3.1.5"><label>śā.1.5</label> tatra, buddhīndriyāṇāṃ śabdādayo viṣayāḥ;
            karmendriyāṇāṃ yathāsaṅkhyaṃ vacanādānānandavisargaviharaṇāni || </l>

          <l xml:id="SS.3.1.6"><label>śā.1.6</label> avyaktaṃ mahānahaṅkāraḥ pañcatanmātrāṇi cety
            aṣṭau prakṛtayaḥ; śeṣāḥ ṣoḍaśa vikārāḥ || </l>

          <l xml:id="SS.3.1.7"><label>śā.1.7</label> svaḥ svaś caiṣāṃ viṣayo 'dhibhūtaṃ; svayam
            adhyātmaṃ; <ref rend="superscript footnote marker">1</ref>adhidaivataṃ<note
              type="footnote"> 'adhidaivatam atha 'iti pā+</note> tu---budherbrahyā,
            ahaṅkārasyeśvaraḥ, manasaś candramāḥ, diśaḥ śrotrasya, tvaco vāyuḥ, sūryaś cakṣuṣaḥ,
            rasanasyāpaḥ, pṛthivī ghrāṇasya, <ref rend="superscript footnote marker">1</ref>vāco
              'gniḥ<note type="footnote">'vacaso 'gniḥ'iti pā+</note>, hastayor indraḥ, pādayor
            viṣṇuḥ, pāyor mitraḥ, prajāpatir upasthasyeti || </l>

          <l xml:id="SS.3.1.8"><label>śā.1.8</label> tatra sarva evācetana eṣa vargaḥ, puruṣaḥ
            pañcaviṃśatitamaḥ kāryakāraṇasaṃyuktaś cetayitā bhavati | saty apy acaitanye pradhānasya
            puruṣakaivalyārthaṃ pravṛttim upadiśanti kṣīrādīṃś cātra hetūn udāharanti || </l>

          <l xml:id="SS.3.1.9"><label>śā.1.9</label> ata ūrdhvaṃ prakṛtipuruṣayoḥ
            sādharmyavaidharmye vyākhyāsyāmaḥ | tadyathā --- ubhāv apy anādī, ubhāv apy anantau,
            ubhāv apy aliṅgau, ubhāv api nityau, <ref rend="superscript               footnote
              marker">1</ref>ubhāv apy anaparau<note type="footnote">'ubhāv apy aparau 'iti
              pā+</note>, ubhau ca sarvagatāv iti; ekā tu prakṛtir acetanā triguṇā bījadharmiṇī
            prasavadharmiṇy amadhyasthadharmiṇī ceti, bahavas tu puruṣāś cetanāvanto 'guṇā
            abījadharmāṇo 'prasavadharmāṇo madhyasthadharmāṇaś ceti || </l>

          <l xml:id="SS.3.1.10"><label>śā.1.10</label> tatra kāraṇānurūpaṃ kāryam iti kṛtvā sarva
            evaite viśeṣāḥ sattvarajastamomayā bhavanti; tadañjanatvāt tanmayatvāc ca tadguṇā eva
            puruṣā bhavantīty eke bhāṣante || </l>

          <l xml:id="SS.3.1.11"><label>śā.1.11</label> vaidyake tu --- svabhāvam īśvaraṃ kālaṃ
            yaddacchāṃ niyatiṃ tathā || pariṇāmaṃ ca manyante prakṛtiṃ pṛthudarśinaḥ || </l>

          <l xml:id="SS.3.1.12"><label>śā.1.12</label> tanmayāny eva bhūtāni tadguṇāny eva cādiśet
            || taiś ca tallakṣaṇaḥ kṛtsno bhūtagrāmo vyajanyata || </l>

          <l xml:id="SS.3.1.13"><label>śā.1.13</label> tasyopayogo 'bhihitaś cikitsāṃ prati sarvadā
            || bhūtebhyo hi paraṃ yasmān nāsti cintā cikitsite || </l>

          <l xml:id="SS.3.1.14"><label>śā.1.14</label> yato 'bhihitaṃ --- tat sambhavadravyasamūho
            bhūtādiruktaḥ; bhautikāni cendriyāṇy āyurvede varṇyante, tathendriyārthaḥ || </l>

          <l xml:id="SS.3.1.15"><label>śā.1.15</label> bhavati cātra--- indriyeṇendiyārthaṃ tu svaṃ
            svaṃ gṛhṇāti mānavaḥ || niyataṃ tulyayonitvānnānyenānyam iti sthitiḥ || </l>

          <l xml:id="SS.3.1.16"><label>śā.1.16</label> na cāyurvedaśāstreṣūpadiśyante sarvagatāḥ
            kṣetrajñā nityāśca; asarvagateṣu ca kṣetrajñeṣu nityapuruṣakhyāpakān hetūn udāharanti;
              <ref rend="superscript footnote               marker">1</ref>āyurvedaśāstrasiddhānteṣv
            asarvagatāḥ kṣetrajñā nityāś ca, tiryagyonimānuṣadeveṣu saṃcaranti dharmādharmanimittaṃ;
            ta ete 'numānagrāhyāḥ paramasūkṣmāścetanāvantaḥ śāśvatā lohitaretasoḥ sannipāteṣv
            abhivyajyante, yato 'bhihitaṃ --- pañcamahābhūtaśarīrisamavāyaḥ puruṣa iti; sa eṣa(eva?)
            karmapuruṣaś cikitsādhikṛtaḥ || </l>

          <l xml:id="SS.3.1.17"><label>śā.1.17</label> tasya sukhaduḥkhe icchādveṣau prayatnaḥ
            prāṇāpānāv unmeṣanimeṣau buddhir manaḥ saṅkalpo vicāraṇā smṛtir vijñānam adhyavasāyo
            viṣayopalabdhiś ca guṇāḥ || </l>

          <l xml:id="SS.3.1.18"><label>śā.1.18</label> sāttvikās tu --- ānṛśaṃsyaṃ saṃvibhāgarucitā
            titikṣā satyaṃ dharma āstikyaṃ jñānaṃ buddhir medhā smṛtir dhṛtir anabhiṣaṅgaś ca;
            rājasās tu --- duḥkhabahulatā'ṭanaśīlatā 'dhṛtirahaṅkāra ānṛtikatvam akāruṇyaṃ dambho
            māno harṣaḥ krodhaś ca; tāmasās tu --- viṣāditvaṃ nāstikyam adharmaśīlatā buddher nidho
            'jñānaṃ durmedhastvam akarmaśīlatā nidrālutvaṃ ceti || </l>

          <l xml:id="SS.3.1.19"><label>śā.1.19</label> āntarikṣāstu---śabdaḥ śabdendriyaṃ
            sarvacchidrasamūho viviktatā ca; vāyavyās tu --- sparśaḥ sparśondriyaṃ sarvaceṣṭāsamūhaḥ
            sarvaśarīraspandanaṃ laghutā ca; taijasās tu --- rūpaṃ rūpendriyaṃ varṇaḥ santāpo
            bhrājiṣṇutā paktir amarṣas taikṣṇyaṃ śauryaṃ ca; āpyās tu --- raso rasanendriyaṃ
            sarvadravasamūho gurutā śaityaṃ sneho retaś ca; pārthivās tu --- gandho gandhendriyaṃ
            sarvamūrtasamūho gurutā ceti || </l>

          <l xml:id="SS.3.1.20"><label>śā.1.20</label> tatra sattvabahulam ākāśaṃ, rajobahulo vāyuḥ,
            sattvarajobahulo 'gniḥ, sattvatamobahulā āpaḥ, tamobahulā pṛthivīti || </l>

          <l xml:id="SS.3.1.21"><label>śā.1.21</label> ślokau cātra bhavataḥ |</l>

          <lg>
            <l xml:id="SS.3.1.21ab">anyonyānupraviṣṭāni sarvāṇy etāni nirdiśet |</l>

            <l xml:id="SS.3.1.21cd">sve sve dravye tu sarveṣāṃ vyaktaṃ lakṣaṇamiṣyate ||</l>








          </lg>
          <l xml:id="SS.3.1.22"><label>śā.1.22</label> aṣṭau prakṛtayaḥ proktā vikārāḥ ṣoḍaśaiva tu
            || kṣetrajñaś ca samāsena svatantraparatantrayoḥ||</l>

          <l xml:id="SS.3.2.2"><label>śā.2.2</label> yathovāca bhagavān dhanvantariḥ || </l>

          <l xml:id="SS.3.2.3"><label>śā.2.3</label> vātapittaśleṣmaśoṇita-<ref rend="superscript

            samarthā bhavanti || </l>

          <l xml:id="SS.3.2.4"><label>śā.2.4</label> teṣu vātavarṇavedanaṃ vātena, pittavarṇavedanaṃ
            pittena, śleṣmavarṇavedanaṃ śleṣmaṇā, śoṇitavarṇavedanaṃ kuṇapagandhyanalpaṃ ca raktena,
            granthibhūtaṃ śleṣmavātābhyāṃ, pūtipūyanibhaṃ pittaśleṣmabhyāṃ, kṣīṇaṃ prāguktaṃ
            pittamārutābhyāṃ, mūtrapurīṣagandhi sannipāteneti | teṣu
            kuṇapagranthipūtipūyakṣīṇaretasaḥ kṛcchrasādhyāh, mūtrapurīṣaretasastvasādhyāḥ (<ref

          <l xml:id="SS.3.2.5"><label>śā.2.5</label> ārtavam api tribhir doṣaiḥ śoṇitacaturthaih

            | teṣu kuṇapagranthipūtipūyakṣīṇamūtrapurīṣaprakāśamasādhyaṃ sādhyamanyacceti || </l>

          <l xml:id="SS.3.2.6"><label>śā.2.6</label> bhavanti cātra | teṣvādyān śukradoṣāṃstrīn
            snehasvedādibhir jayet || kriyāviśeṣairmatimāṃstathā cottarav(b?)astibhiḥ || </l>

          <l xml:id="SS.3.2.7"><label>śā.2.7</label> pāyayeta naraṃ sarpirbhiṣak kuṇaparetasi ||
            dhātakīpuṣpakhadiradāḍimārjunasādhitam || </l>

          <l xml:id="SS.3.2.8"><label>śā.2.8</label> pāyatedathavā sarpiḥ śālasārādisādhitam ||
            granthibhūte <ref rend="superscript footnote marker">1</ref>śaṭīsaiddhaṃ<note
              type="footnote">''śmabhitsiddhaṃ'iti pā+</note> pālāśe vā'pi bhasmani || </l>

          <l xml:id="SS.3.2.9"><label>śā.2.9</label> parūṣakavaṭādibhyāṃ pūyaprakhye ca sādhitam ||
            prāguktaṃ vakṣyate yac ca tat kāryaṃ kṣīṇaretasi || </l>

          <l xml:id="SS.3.2.10"><label>śā.2.10</label> viṭprabhe pāyayet siddhaṃ <ref
              rend="superscript footnote marker">1</ref>citrakośīrahiṅgubhiḥ || [<ref
              rend="superscript footnote marker">1</ref>asyāgre 'snehādiś ca kramaḥ kāryaḥ

            nirūḍhamanuvāsitam || </l>

          <l xml:id="SS.3.2.11"><label>śā.2.11</label> yojayecchukradoṣārtaṃ samyaguttarabastinā ||

          <l xml:id="SS.3.2.12"><label>śā.2.12</label> śukramicchanti kecittu tailakṣaudranibhaṃ
            tathā || vidhimuttarabastyantaṃ kuryād ārtavaśuddhaye || </l>

          <l xml:id="SS.3.2.13"><label>śā.2.13</label> strīṇāṃ snehādiyuktānāṃ catasṛṣvārtavārtiṣu

          <l xml:id="SS.3.2.14"><label>śā.2.14</label> granthibhūte pibet pāṭhāṃ tryūṣaṇaṃ vṛkṣakāṇi

            'iti pā+] </l>

          <l xml:id="SS.3.2.15"><label>śā.2.15</label> pibedbhadraśriyaḥ kvāthaṃ candanakvātham eva

          <l xml:id="SS.3.2.16"><label>śā.2.16</label>
            <ref rend="superscript footnote marker">1</ref>yogānāṃ<note type="footnote">'doṣāṇāṃ'iti

          <l xml:id="SS.3.2.17"><label>śā.2.17</label> śaśāsṛkpratimaṃ yat tu yad vā lākṣārasopamam
            || tadārtavaṃ praśaṃsanti yad vāso na virañjayet || </l>

          <l xml:id="SS.3.2.18"><label>śā.2.18</label> tad evātiprasaṅgena pranṛttamanṛtāvapi ||

              'nyad raktalakṣaṇāt[<ref rend="superscript               footnote marker"
            >1</ref>'vijānīyād uktaṃ lakṣaṇalakṣitam'iti pā+] || </l>

          <l xml:id="SS.3.2.19"><label>śā.2.19</label> asṛgdaro bhavet sarvaḥ sāṅgamardaḥ savedanaḥ

          <l xml:id="SS.3.2.20"><label>śā.2.20</label> dāhaḥ pralāpaḥ pāṇḍutvaṃ tandrā rogāś ca
