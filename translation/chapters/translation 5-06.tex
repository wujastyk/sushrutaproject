         p.\,\pageref{water-detox} above.}
     \gls{udumbara}, 
     \gls{karaghāṭaka}, 
     \gls{arjuna},
     \gls{sarjja}, 
     \gls{kapītana}, 
     \gls{śleṣmātakā}, 
     \gls{aṅkoṭha},
     \gls{kuṭaja},
     \gls{śamī},
