          <l xml:id="SS.2.1.11"> yathā 'gniḥ pañcadhā bhinno nāmasthānakriyāmayaiḥ | bhinno 'nilas
            tasthā hy eko nāmasthānakriyāmayaiḥ || </l>

          <l xml:id="SS.2.1.12"> prāṇodānau samānaś ca vyānaś cāpāna eva ca | sthānasthā mārutāḥ `
            yāpayanti śarīriṇam || </l>

          <l xml:id="SS.2.1.13"> yo vāyur vaktrasaṃcārī sa prāṇo nāma dehadhṛk | so 'nnaṃ
            praveśayaty antaḥ prāṇāṃś cāpy avalambate || </l>

          <l xml:id="SS.2.1.14"> prāyaśaḥ kurute duṣṭo hikkāśvāsādikān gadān || udāno nāma yas
            tūrdhvam upaiti pavanottamaḥ || </l>

          <l xml:id="SS.2.1.15"> tena bhāṣitagītādiviśeṣo 'bhipravartate | ūrdhvajatrugatān rogān
            karoti ca viśeṣataḥ || </l>


          <l xml:id="SS.2.1.16"> āmapakvāśayacaraḥ samāno vahnisaṅgataḥ | so 'nnaṃ pacati tajjāṃś ca
            viśeṣān vivinakti hi || </l>

          <l xml:id="SS.2.1.17"> gulmāgni<pc> </pc>sādātīsāraprabhṛtīn kurute gadān | kṛtsnadehacaro
            vyāno rasasaṃvahanodyataḥ || </l>

          <l xml:id="SS.2.1.18"> svedāsṛksrāvaṇaś cāpi pañcadhā ceṣṭayaty api | kruddhaś ca kurute
            rogān prāyaśaḥ sarvadehagān || </l>

          <l xml:id="SS.2.1.19"> pakvādhānālayo 'pānaḥ kāle karṣati cāpy adhaḥ | samīraṇaḥ
            śakṛnmūtraṃ śukragarbhārtavāni ca || </l>

          <l xml:id="SS.2.1.20"> kruddhaś ca kurute rogān ghorān bastigudāśrayān | śukradoṣapramehās
            tu vyānāpānaprakopajāḥ || </l>

          <l xml:id="SS.2.1.21ab"> yugapat kupitāś cāpi dehaṃ bhindyur asaṃśayam |</l>

          <l xml:id="SS.2.1.21cd"> ata ūrdhvaṃ pravakṣyāmi nānā<pc> </pc>sthānāntarāśritaḥ || </l>

          <l xml:id="SS.2.1.23"> mohaṃ mūrcchāṃ pipāsāṃ ca hṛdgrahaṃ pārśvavedanām | pakvāśayastho

          <l xml:id="SS.2.1.24"> kṛcchramūtrapurīṣatvam ānāhaṃ trikavedanām | śrotrādiṣv

          <l xml:id="SS.2.1.30"> hastapādaśirodhātūṃs tathā saṃcarati kramāt | vyāpnuyād vākhilaṃ

          <l xml:id="SS.2.1.36"> asvedaharṣau mando 'gniḥ śītastambhau kaphāvṛte | samāne

            ceti || </l>

          <l xml:id="SS.2.16.37"> jihvā 'nilena sphuṭitā prasuptā bhavec ca śākacchadanaprakāśā |
            <caesura/>
             pittena pītā paridahyate ca citā saraktair api kaṇṭakaiś ca |
            <caesura/>
             kaphena gurvī bahalā citā ca māṃsodgamaiḥ śālmalikaṇṭakābhaiḥ || </l>

          <l xml:id="SS.2.16.38"> jihvātale yaḥ śvayathuḥ pragāḍhaḥ so 'lāsasaṃjñaḥ kapharaktamūrtiḥ
            |
            <caesura/>
             jihvāṃ sa tu stambhayati pravṛddho mūle tu jihvā bhṛśam eti pākam || </l>

          <l xml:id="SS.2.16.39"> jihvāgrarūpaḥ śvayathur hi jihvām unnamya jātaḥ kapharaktayoniḥ |
            <caesura/>
             prasekakaṇḍūparidāhayuktā prakathyate 'sāv upajihviketi || </l>

          <l xml:id="SS.2.16.40"> tālugatās tu galaśuṇḍikā, tuṇḍikerī, adhruṣaḥ, kacchapaḥ, arbudaṃ,
            māṃsasaṅghātaḥ, tālupuppuṭaḥ, tāluśoṣaḥ, tālupāka iti || </l>

          <l xml:id="SS.2.16.41"> śleṣmāsṛgbhyāṃ tālumūlāt pravṛddho dīrghaḥ śopho
            dhmātabastiprakāśaḥ |
            <caesura/>
             tṛṣṇākāsaśvāsakṛt saṃpradiṣṭo vyādhir vaidyaiḥ kaṇṭhaśuṇḍīti nāmnā || </l>

          <l xml:id="SS.2.16.42"> śophaḥ stabdho lohiṃtastāludeśe raktājñeyaḥ so 'dhruṣo
            rugjvarāḍhyaḥ || </l>

          <l xml:id="SS.2.16.43"> kūrmotsanno 'vedano 'śīghrajanmā 'rakto jñeyaḥ kacchapaḥ śleṣmaṇā
            syāt |
            <caesura/>
             padmākāraṃ tālumadhye tu śophaṃ vidyād raktād arbudaṃ proktaliṅgam || </l>

          <l xml:id="SS.2.16.44"> duṣṭaṃ māṃsaṃ śleṣmaṇā nīrujaṃ ca tālvantaḥsthaṃ māṃsasaṅghātam
            āhuḥ |
            <caesura/>
             nīruk sthāyī kolamātraḥ kaphāt syān medoyuktāt puppuṭastāludeśe || </l>

          <l xml:id="SS.2.16.45"> śoso 'tyarthaṃ dīryate cāpi tāluḥ śvāso vātāttāluśoṣaḥ sapittāt |
            <caesura/>
             pittaṃ kuryāt pākam atyarthaghoraṃ tāluny enaṃ tālupākaṃ vadanti || </l>

          <l xml:id="SS.2.16.46"> kaṇṭhagatās tu rohiṇyaḥ pañca, kaṇṭhaśālūkam, adhijihvo, balayo,
            balāsa, ekavṛndo, vṛndaḥ, śataghnī, gilāyuḥ, galavidradhiḥ, galaughaḥ, svaraghno,
            māṃsatāno, vidārī ceti || </l>

          <l xml:id="SS.2.16.47"> gale 'nilaḥ pittakaphau ca mūrcchitau pṛthak samastāś ca tathaiva
            śoṇitam |
            <caesura/>
             pradūṣya māṃsaṃ galarodhino 'ṅkurān sṛjanti yān sā 'suharā hi rohiṇī || </l>

          <l xml:id="SS.2.16.48"> jihvāṃ samantād bhṛśavedanā ye māṃsāṅkurāḥ kaṇṭhanirodhinaḥ syuḥ |
            <caesura/>
             tāṃ rohiṇīṃ vātakṛtāṃ vadanti vātātm akopadravagāḍhayuktām || </l>

          <l xml:id="SS.2.16.49"> kṣiprodgamā kṣipravidāhapākā tīvrajvarā pittanimittajā syāt |
            <caesura/>
             srotonirodhiny api mandapākā gurvī sthirā sā kaphasaṃbhavā vai || </l>

          <l xml:id="SS.2.16.50"> gambhīrapākā 'prativāravīryā tridoṣaliṅgā trayasaṃbhavā syāt |
            <caesura/>
             sphoṭācitā pittasamānaliṅgā 'sādhyā pradiṣṭā rudhirātmikeyam || </l>

          <l xml:id="SS.2.16.51"> kolāsthimātraḥ kaphasṃbhavo yo granthir gale kaṇṭakaśūkabhūtaḥ |
            <caesura/>
             kharaḥ sthiraḥ śastranipātasādhyas taṃ kaṇṭhaśālūkam iti bruvanti || </l>

          <l xml:id="SS.2.16.52"> jihvāgrarūpaḥ śvayathuḥ kaphāttu jihvāprabandhopari raktamiśrāt |
            <caesura/>
             jñeyo 'dhijihvaḥ khalu roga eṣa vivarjayed āgatapākam enam || </l>

          <l xml:id="SS.2.16.53"> balāsa evāyatamunnataṃ ca śophaṃ karotyannagatiṃ nivārya |
            <caesura/>
             taṃ sarvathaivāprativāravīryaṃ vivarjanīyaṃ valayaṃ vadanti || </l>

          <l xml:id="SS.2.16.54"> gale tu śophaṃ kurutaḥ pravṛddhau śleṣmānilau śvāsarujopapannam |
            <caesura/>
             marmacchidaṃ dustaram etad āhur balāsasṃjñaṃ nipuṇā vikāram || </l>

          <l xml:id="SS.2.16.55"> vṛttonnato yaḥ śvayathuḥ sadāhaḥ kaṇḍvanvito 'pākyamṛdurguruś ca |
            <caesura/>
             nāmnaikavṛndaḥ parikīrtito 'sau vyādhir balāsakṣatajaprasūtaḥ || </l>

          <l xml:id="SS.2.16.56"> samunnataṃ vṛttam amandadāhaṃ tīvrajvaraṃ vṛndamudāharanti |
            <caesura/>
             taṃ cāpi pittakṣatajaprakopādvidyāt satodaṃ pavanāsrajaṃ tu || </l>

          <l xml:id="SS.2.16.57"> vartir ghanā kaṇṭhanirodhinī yā citā 'timātraṃ piśitaprarohaiḥ |
            <caesura/>
             nānārujocchrāyakarī tridoṣājjñeyā śataghnīva śataghny asādhyā || </l>

          <l xml:id="SS.2.16.58"> granthirgale tvāmalakāsthimātraḥ sthiro 'lparuk syāt
            kapharaktamūrtiḥ |
            <caesura/>
             saṃlakṣyate saktam ivāśanaṃ ca sa śastrasādhyas tu gilāyusaṃjñaḥ || </l>

          <l xml:id="SS.2.16.59"> sarvaṃ galaṃ vyāpya samutthito yaḥ śopho rujo yatra ca santi
            sarvāḥ |
            <caesura/>
             sa sarvadoṣo galavidradhis tu tasyaiva tulyaḥ khalu sarvajasya || </l>

          <l xml:id="SS.2.16.60"> śopho mahānannajalāvarodhī tīvrajvaro vātagaternihantā |
            <caesura/>
             kaphena jāto rudhirānvitena gale galaughaḥ parikīrtyate 'sau || </l>

          <l xml:id="SS.2.16.61"> yo 'tipratāmyan śvasiti prasaktaṃ bhinnasvaraḥ śuṣkavimuktakaṇṭhaḥ
            |
            <caesura/>
             kaphopadigdheṣv anilāyaneṣu jñeyaḥ sa rogaḥ śvasanāt svaraghnaḥ || </l>

          <l xml:id="SS.2.16.62"> pratānavān yaḥ śvayathuḥ sukaṣṭo galoparodhaṃ kurute krameṇa |
            <caesura/>
             sa māṃsatānaḥ kathito 'valambī prāṇapraṇut sarvakṛto vikāraḥ || </l>

          <l xml:id="SS.2.16.63"> sadāhatodaṃ śvayathuṃ saraktam antargale pūtiviśīrṇamāṃsam |
            <caesura/>
             pittena vidyād vadane vidārīṃ pārśve viśeṣāt sa tu yena śete || </l>

          <l xml:id="SS.2.16.64"> sarvasarās tu vātapittakaphaśoṇitanimittāḥ || </l>

          <l xml:id="SS.2.16.65"> sphoṭaiḥ satodair vadanaṃ samantād yasyācitaṃ sarvasaraḥ sa vātāt
            |
            <caesura/>
             raktaiḥ sadāhais tanubhiḥ sapītair yasyācitaṃ cāpi sa pittakopāt || </l>

          <l xml:id="SS.2.16.66"> kaṇḍūyutair alparujaiḥ sarvaṇair yasyācitaṃ cāpi sa vai kaphena |
            <caesura/>
             raktena pittodita eka eva kaiś cit pradiṣṭo mukhapākasaṃjñaḥ ||</l>

          <l xml:id="SS.2.16.trailer">iti suśrutasaṃhitāyāṃ nidānasthāne mukharoganidānaṃ nāma
            ṣoḍaśo 'dhyāyaḥ ||16 ||</l>
